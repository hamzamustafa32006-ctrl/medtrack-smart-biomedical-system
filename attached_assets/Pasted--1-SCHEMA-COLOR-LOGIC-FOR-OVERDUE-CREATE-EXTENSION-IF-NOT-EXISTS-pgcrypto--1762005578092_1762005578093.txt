CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- حقول جديدة مع قيم افتراضية
ALTER TABLE equipment
  ADD COLUMN IF NOT EXISTS criticality TEXT DEFAULT 'Medium',          -- Low/Medium/High
  ADD COLUMN IF NOT EXISTS condition    TEXT DEFAULT 'Good',            -- Good/Fair/Poor
  ADD COLUMN IF NOT EXISTS priority     TEXT DEFAULT 'Normal',          -- Normal/Urgent
  ADD COLUMN IF NOT EXISTS risk_score   INT  DEFAULT 0,
  ADD COLUMN IF NOT EXISTS status_color TEXT DEFAULT 'green',           -- green/orange/red
  ADD COLUMN IF NOT EXISTS last_check   TIMESTAMP DEFAULT now();

-- حقلين مشتقين نخزنهم لتسهيل العرض
ALTER TABLE equipment
  ADD COLUMN IF NOT EXISTS days_overdue INT,
  ADD COLUMN IF NOT EXISTS is_overdue   BOOLEAN;

-- تحديث القيم حسب التواريخ الحالية
UPDATE equipment
SET
  days_overdue = GREATEST((CURRENT_DATE - next_due_date)::int, 0),
  is_overdue   = (next_due_date < CURRENT_DATE),
  status_color = CASE
                   WHEN next_due_date < CURRENT_DATE - INTERVAL '30 days' THEN 'red'      -- متأخر أكثر من 30 يوم
                   WHEN next_due_date < CURRENT_DATE THEN 'orange'                        -- متأخر 1-30 يوم
                   WHEN next_due_date <= CURRENT_DATE + INTERVAL '7 days' THEN 'orange'   -- قرب موعده خلال أسبوع
                   ELSE 'green'
                 END,
  priority     = CASE
                   WHEN next_due_date < CURRENT_DATE THEN 'Urgent'
                   ELSE 'Normal'
                 END,
  criticality  = CASE
                   WHEN type ILIKE '%Life Support%' OR department ILIKE '%ICU%' THEN 'High'
                   WHEN department ILIKE '%ER%' OR department ILIKE '%OR%' THEN 'High'
                   WHEN department ILIKE '%Radiology%' THEN 'Medium'
                   ELSE 'Low'
                 END;

-- إعادة حساب الألوان للأجهزة المضافة الآن

SET
  days_overdue = GREATEST((CURRENT_DATE - next_due_date)::int, 0),
  is_overdue   = (next_due_date < CURRENT_DATE),
  status_color = CASE
                   WHEN next_due_date < CURRENT_DATE - INTERVAL '30 days' THEN 'red'

                   ELSE 'green'
                 END;

-- ملخّص مجمّع لعدادات الألوان (View) — سهل على الـ API
CREATE OR REPLACE VIEW equipment_color_summary AS
SELECT
  COUNT(*)::int                                         AS total,
  SUM(CASE WHEN status_color='red'    THEN 1 ELSE 0 END)::int AS red,
  SUM(CASE WHEN status_color='orange' THEN 1 ELSE 0 END)::int AS orange,
  SUM(CASE WHEN status_color='green'  THEN 1 ELSE 0 END)::int AS green,
  SUM(CASE WHEN is_overdue THEN 1 ELSE 0 END)::int           AS overdue
FROM equipment;
