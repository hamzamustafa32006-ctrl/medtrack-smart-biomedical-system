## 🧩 2. **Dashboard Container (Main Logic)**

```tsx
import { useEffect, useState } from "react";
import DashboardHeader from "./DashboardHeader";
import DashboardWidgets from "./DashboardWidgets";
import EquipmentCharts from "./EquipmentCharts";
import EquipmentTable from "./EquipmentTable";

export default function DashboardContainer() {
  const [filters, setFilters] = useState({ status: "", facility: "", search: "" });
  const [equipment, setEquipment] = useState([]);
  const [loading, setLoading] = useState(false);

  // Fetch data dynamically
  const fetchEquipmentData = async (newFilters) => {
    setLoading(true);
    const params = new URLSearchParams();
    if (newFilters.status) params.append("status", newFilters.status);
    if (newFilters.facility) params.append("facility", newFilters.facility);
    if (newFilters.search) params.append("search", newFilters.search);

    const res = await fetch(`/equipment?${params.toString()}`);
    const data = await res.json();
    setEquipment(data);
    setLoading(false);
  };

  useEffect(() => {
    fetchEquipmentData(filters);
  }, [filters]);

  return (
    <div className="p-6 space-y-6">
      <DashboardHeader filters={filters} onFilterChange={setFilters} />
      <DashboardWidgets equipment={equipment} loading={loading} />
      <EquipmentCharts equipment={equipment} />
      <EquipmentTable equipment={equipment} loading={loading} />
    </div>
  );
}
```

✅ Centralized control — one source of truth for all dashboard data
✅ Every filter change triggers live updates across widgets & charts

## 📊 4. **Real-Time KPI Cards**

```tsx
import { Card, CardContent } from "@/components/ui/card";

export default function DashboardWidgets({ equipment, loading }) {
  const stats = {
    total: equipment.length,
    active: equipment.filter((e) => e.status === "Active").length,
    maintenance: equipment.filter((e) => e.status === "Under Maintenance").length,
    overdue: equipment.filter((e) => e.status === "Overdue").length,
  };

  return (
    <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
      <StatCard title="Total Equipment" value={stats.total} color="text-blue-600" />
      <StatCard title="Active" value={stats.active} color="text-green-600" />
      <StatCard title="Under Maintenance" value={stats.maintenance} color="text-yellow-600" />
      <StatCard title="Overdue" value={stats.overdue} color="text-red-600" />
    </div>
  );
}

function StatCard({ title, value, color }) {
  return (
    <Card>
      <CardContent className="p-4">
        <h3 className="text-sm text-gray-500">{title}</h3>
        <p className={`text-2xl font-semibold ${color}`}>{value}</p>
      </CardContent>
    </Card>
  );
}
```

✅ Auto-refreshes with every filter change
✅ Perfect for quick at-a-glance insight

## 📈 5. **Interactive Charts (with Recharts)**

```tsx
import { PieChart, Pie, Cell, Tooltip, Legend } from "recharts";

export default function EquipmentCharts({ equipment }) {
  const data = [
    { name: "Active", value: equipment.filter((e) => e.status === "Active").length },
    { name: "Under Maintenance", value: equipment.filter((e) => e.status === "Under Maintenance").length },
    { name: "Overdue", value: equipment.filter((e) => e.status === "Overdue").length },
  ];
  const COLORS = ["#22c55e", "#eab308", "#ef4444"];

  return (
    <div className="flex justify-center items-center mt-6">
      <PieChart width={350} height={250}>
        <Pie
          data={data}
          cx="50%"
          cy="50%"
          labelLine={false}
          label={({ name, value }) => `${name}: ${value}`}
          outerRadius={100}
          fill="#8884d8"
          dataKey="value"
        >
          {data.map((entry, index) => (
            <Cell key={index} fill={COLORS[index % COLORS.length]} />
          ))}
        </Pie>
        <Tooltip />
        <Legend />
      </PieChart>
    </div>
  );
}
```

✅ Updates dynamically based on current filters
✅ Works great for status or facility-based data

## 🧰 6. **Equipment Table (Filtered Data)**

```tsx
export default function EquipmentTable({ equipment, loading }) {
  if (loading) return <p className="text-gray-500">Loading data...</p>;
  if (!equipment.length) return <p className="text-gray-500">No equipment found.</p>;

  return (
    <div className="overflow-x-auto border rounded-lg">
      <table className="min-w-full text-sm">
        <thead className="bg-gray-100">
          <tr>
            <th className="px-4 py-2 text-left">Name</th>
            <th className="px-4 py-2 text-left">Facility</th>
            <th className="px-4 py-2 text-left">Status</th>
            <th className="px-4 py-2 text-left">Next Due</th>
          </tr>
        </thead>
        <tbody>
          {equipment.map((e) => (
            <tr key={e.id} className="border-t hover:bg-gray-50">
              <td className="px-4 py-2">{e.name}</td>
              <td className="px-4 py-2">{e.facility_name}</td>
              <td className="px-4 py-2">{e.status}</td>
              <td className="px-4 py-2">{e.next_due_date}</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  );
}
```

✅ Auto-refreshes live on every filter or search
✅ Clean, responsive layout
