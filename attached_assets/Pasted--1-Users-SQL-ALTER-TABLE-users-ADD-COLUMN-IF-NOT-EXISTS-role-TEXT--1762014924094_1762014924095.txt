الخطوة 1: تعديل جدول المستخدمين (Users)
🧩 SQL
ALTER TABLE users
ADD COLUMN IF NOT EXISTS role TEXT CHECK (role IN ('Admin', 'Technician', 'Supervisor', 'Viewer')) DEFAULT 'Technician';

CREATE TABLE IF NOT EXISTS roles_permissions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  role TEXT NOT NULL,
  permission TEXT NOT NULL
);

🧩 الخطوة 4: تعديل الـ API للتحقق من الصلاحيات
🔹 مثال (Node.js / Express):
async function authorize(userId, permission) {
  const result = await db.query(`
    SELECT 1 FROM roles_permissions
    WHERE role = (SELECT role FROM users WHERE id = $1)
    AND permission = $2
  `, [userId, permission]);
  return result.rowCount > 0;
}

// مثال استخدام
app.post('/maintenance-records', async (req, res) => {
  const { userId } = req.body;
  const allowed = await authorize(userId, 'edit_all');
  if (!allowed) return res.status(403).json({ error: 'Access denied' });
  
  // باقي الكود لإنشاء السجل...
});
💡 هذا يضمن أن فقط المستخدمين المصرح لهم يمكنهم التعديل أو الحذف.
