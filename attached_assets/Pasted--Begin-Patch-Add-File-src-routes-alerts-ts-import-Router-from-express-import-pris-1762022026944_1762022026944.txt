+import { Router } from 'express';
+import { prisma } from '../prisma';
+import { addDays, startOfWeek, endOfWeek } from 'date-fns';
+
+// نافذة الـ Upcoming شهر (تقدر تغيّرها من env)
+const UPCOMING_WINDOW_DAYS = Number(process.env.UPCOMING_WINDOW_DAYS || 30);
+// بداية الأسبوع السبت (6) بما يناسب الخليج؛ غيّرها إذا تبي الأحد=0
+const WEEK_STARTS_ON: 0|1|2|3|4|5|6 = Number(process.env.WEEK_STARTS_ON ?? 6) as any;
+
+const router = Router();
+
+/**
+ * تعريفات منطق العد:
+ * - Critical Alerts (أجهزة): أي جهاز alert_level ∈ {Urgent, Critical} أو حالته Overdue
+ * - Upcoming Maintenance (أجهزة): next_due_date بين الآن و +UPCOMING_WINDOW_DAYS
+ * - Resolved This Week (أجهزة): أجهزة لها maintenance_record مكتمل داخل أسبوعنا الحالي
+ *
+ * أسماء الحقول القابلة للتعديل حسب سكيمتك:
+ * equipment: id, alert_level, status, next_due_date, is_active, deleted_at
+ * maintenance_record: id, equipment_id, status ('completed'), completed_at
+ */
+router.get('/summary', async (_req, res) => {
+  try {
+    const now = new Date();
+    const upcomingTo = addDays(now, UPCOMING_WINDOW_DAYS);
+    const weekStart = startOfWeek(now, { weekStartsOn: WEEK_STARTS_ON });
+    const weekEnd   = endOfWeek(now,   { weekStartsOn: WEEK_STARTS_ON });
+
+    // 1) Critical (أجهزة)
+    const critical = await prisma.equipment.count({
+      where: {
+        // لا تعد الأجهزة المحذوفة/المتقاعدة إن وجدت لديك مثل هذه الحقول
+        deleted_at: null,
+        // active/operational إن كان عندك فلتر مماثل
+        // is_active: true,
+        OR: [
+          { alert_level: { in: ['Urgent', 'Critical'] } },
+          { status: 'overdue' },
+          { next_due_date: { lt: now } }, // احتياط لو ما تستخدم status=overdue
+        ],
+      },
+    });
+
+    // 2) Upcoming (أجهزة خلال شهر)
+    const upcoming = await prisma.equipment.count({
+      where: {
+        deleted_at: null,
+        // is_active: true,
+        next_due_date: { gte: now, lte: upcomingTo },
+        // غير مكتملة/ما تزال بحاجة صيانة (اختياري)
+        // status: { not: 'completed' },
+      },
+    });
+
+    // 3) Resolved This Week (أجهزة لها قيود صيانة اكتملت هذا الأسبوع)
+    const resolvedThisWeekRaw = await prisma.maintenance_record.findMany({
+      where: {
+        status: 'completed',
+        completed_at: { gte: weekStart, lte: weekEnd },
+      },
+      select: { equipment_id: true },
+    });
+    // مميز الأجهزة حتى لو أكثر من سجل لنفس الجهاز
+    const resolvedThisWeek = new Set(resolvedThisWeekRaw.map(r => r.equipment_id)).size;
+
+    res.json({ critical, upcoming, resolvedThisWeek });
+  } catch (err) {
+    console.error('GET /api/alerts/summary failed', err);
+    res.status(500).json({ error: 'FAILED_TO_COMPUTE_ALERT_SUMMARY' });
+  }
+});
+
+export default router;

+import { useEffect, useState } from 'react';
+
+export type AlertSummary = { critical: number; upcoming: number; resolvedThisWeek: number };
+
+export function useAlertSummary(refreshMs = 60000) {
+  const [data, setData] = useState<AlertSummary>({ critical: 0, upcoming: 0, resolvedThisWeek: 0 });
+  const [loading, setLoading] = useState(true);
+  const [error, setError] = useState<string | null>(null);
+
+  async function load() {
+    try {
+      setError(null);
+      const res = await fetch('/api/alerts/summary', { credentials: 'include' });
+      if (!res.ok) throw new Error(`HTTP ${res.status}`);
+      const json = await res.json();
+      setData(json);
+    } catch (e: any) {
+      setError(e?.message || 'failed');
+    } finally {
+      setLoading(false);
+    }
+  }
+
+  useEffect(() => {
+    let cancel = false;
+    load();
+    const id = setInterval(() => { if (!cancel) load(); }, refreshMs);
+    return () => { cancel = true; clearInterval(id); };
+  }, []);
+
+  return { data, loading, error, reload: load };
+}

+  const v = (n: number) => (loading ? '—' : n);
