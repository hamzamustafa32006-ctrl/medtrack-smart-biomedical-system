## 🧩 2. Frontend Design — Dynamic Filter & Search Component

```tsx
import { useState, useEffect } from "react";

export default function EquipmentDashboardFilters({ onFilterChange }) {
  const [status, setStatus] = useState("");
  const [search, setSearch] = useState("");
  const [facility, setFacility] = useState("");
  const [debounceTimer, setDebounceTimer] = useState(null);

  const handleSearchChange = (e) => {
    const value = e.target.value;
    setSearch(value);

    // Live debounce (auto-search after 500ms)
    clearTimeout(debounceTimer);
    const timer = setTimeout(() => {
      onFilterChange({ status, search: value, facility });
    }, 500);
    setDebounceTimer(timer);
  };

  const handleStatusChange = (e) => {
    setStatus(e.target.value);
    onFilterChange({ status: e.target.value, search, facility });
  };

  const handleFacilityChange = (e) => {
    setFacility(e.target.value);
    onFilterChange({ status, search, facility: e.target.value });
  };

  return (
    <div className="flex flex-wrap gap-3 items-center mb-6">
      {/* Status Filter */}
      <select
        value={status}
        onChange={handleStatusChange}
        className="border px-3 py-2 rounded-lg text-sm focus:ring-2 focus:ring-blue-500"
      >
        <option value="">All Statuses</option>
        <option value="Active">Active</option>
        <option value="Pending Inspection">Pending Inspection</option>
        <option value="Under Maintenance">Under Maintenance</option>
        <option value="Overdue">Overdue</option>
        <option value="Decommissioned">Decommissioned</option>
      </select>

      {/* Facility Filter */}
      <select
        value={facility}
        onChange={handleFacilityChange}
        className="border px-3 py-2 rounded-lg text-sm focus:ring-2 focus:ring-blue-500"
      >
        <option value="">All Facilities</option>
        <option value="AASU Hospital">AASU Hospital</option>
        <option value="Medical Center 2">Medical Center 2</option>
      </select>

      {/* Search Input */}
      <input
        type="text"
        placeholder="Search equipment..."
        value={search}
        onChange={handleSearchChange}
        className="border px-3 py-2 w-64 rounded-lg text-sm focus:ring-2 focus:ring-blue-500"
      />
    </div>
  );
}
```

✅ Supports **debounced search** (instant, no “Apply” button needed)
✅ Updates filters **live** (equipment, analytics, and charts sync together)
✅ Easily extended (e.g., add department filter later)

## ⚙️ 3. Fetch & Update Logic (Dashboard Container)

```tsx
import { useState, useEffect } from "react";
import EquipmentDashboardFilters from "./EquipmentDashboardFilters";
import EquipmentTable from "./EquipmentTable";
import DashboardWidgets from "./DashboardWidgets";
import EquipmentCharts from "./EquipmentCharts";

export default function DashboardPage() {
  const [equipment, setEquipment] = useState([]);
  const [filters, setFilters] = useState({});
  const [stats, setStats] = useState({ total: 0, active: 0, overdue: 0 });

  useEffect(() => {
    fetchData(filters);
  }, [filters]);

  const fetchData = async ({ status, search, facility }) => {
    const params = new URLSearchParams();
    if (status) params.append("status", status);
    if (search) params.append("search", search);
    if (facility) params.append("facility", facility);

    const res = await fetch(`/equipment?${params.toString()}`);
    const data = await res.json();

    setEquipment(data);

    // Auto-update analytics widgets
    const counts = {
      total: data.length,
      active: data.filter((e) => e.status === "Active").length,
      overdue: data.filter((e) => e.status === "Overdue").length,
    };
    setStats(counts);
  };

  return (
    <div className="p-6">
      <h1 className="text-xl font-bold mb-4">Equipment Dashboard</h1>

      {/* Filters */}
      <EquipmentDashboardFilters onFilterChange={setFilters} />

      {/* Widgets */}
      <DashboardWidgets stats={stats} />

      {/* Charts */}
      <EquipmentCharts equipment={equipment} />

      {/* Equipment Table */}
      <EquipmentTable equipment={equipment} />
    </div>
  );
}
```

✅ Automatically refreshes **Equipment Table + Widgets + Charts**
✅ No reloads — uses real-time data from your existing API
✅ Easy to extend to maintenance calendar

  return (
    <PieChart width={400} height={300}>
