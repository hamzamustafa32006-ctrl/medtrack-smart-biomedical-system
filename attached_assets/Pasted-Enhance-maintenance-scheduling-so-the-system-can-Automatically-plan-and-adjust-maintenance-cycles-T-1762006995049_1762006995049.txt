🧱 2. Database & Model Improvements
Add a dedicated maintenance_schedules table — separate from equipment — to give more flexibility.
CREATE TABLE maintenance_schedules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  equipment_id UUID REFERENCES equipment(id),
  maintenance_type TEXT CHECK (maintenance_type IN ('Preventive','Corrective','Calibration','Inspection')) DEFAULT 'Preventive',
  frequency_days INT NOT NULL DEFAULT 180,
  last_maintenance_date DATE,
  next_due_date DATE,
  assigned_to UUID REFERENCES users(id),
  priority TEXT CHECK (priority IN ('Low','Medium','High','Critical')) DEFAULT 'Medium',
  status TEXT CHECK (status IN ('Scheduled','In Progress','Completed','Overdue')) DEFAULT 'Scheduled',
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
✅ Benefits:
Each equipment can have multiple maintenance types (e.g., preventive + calibration).
Easier to report and visualize upcoming schedules per department or facility.
🤖 3. Smart Automation — Schedule Management Engine
✅ A. Automatic Scheduling Job (Daily)
Recalculate next due dates and flag overdue tasks automatically.
import cron from 'node-cron';
import db from './db.js';
import dayjs from 'dayjs';

cron.schedule('0 1 * * *', async () => {  // 1 AM daily
  const today = dayjs().format('YYYY-MM-DD');

  // Fetch all active schedules
  const { rows: schedules } = await db.query(`
    SELECT id, equipment_id, next_due_date, frequency_days, status
    FROM maintenance_schedules
    WHERE status != 'Completed'
  `);

  for (const s of schedules) {
    const daysLeft = dayjs(s.next_due_date).diff(today, 'day');
    let newStatus = s.status;

    if (daysLeft <= 0 && s.status !== 'Overdue') newStatus = 'Overdue';
    else if (daysLeft <= 7 && s.status === 'Scheduled') newStatus = 'In Progress';

    if (newStatus !== s.status) {
      await db.query(
        'UPDATE maintenance_schedules SET status=$1, updated_at=NOW() WHERE id=$2',
        [newStatus, s.id]
      );

      // Optional: trigger a maintenance alert
      await db.query(`
        INSERT INTO maintenance_alerts (equipment_id, facility_id, severity, title, message, due_date)
        SELECT e.id, e.facility_id, 'Warning', 'Maintenance due soon',
        CONCAT(e.name, ' maintenance due in ', $1, ' days.'), $2
        FROM equipment e WHERE e.id=$3
      `, [daysLeft, s.next_due_date, s.equipment_id]);
    }
  }
});
✅ Keeps all maintenance schedules accurate
✅ Automatically updates “In Progress” or “Overdue” status
✅ Creates linked alerts in real time
✅ B. Automatic Rescheduling (When Completed)
When a technician completes a maintenance task:
UPDATE maintenance_schedules
SET
  last_maintenance_date = CURRENT_DATE,
  next_due_date = CURRENT_DATE + (frequency_days || ' days')::interval,
  status = 'Scheduled'
WHERE id = $1;

UPDATE equipment
SET last_maintenance_date = CURRENT_DATE,
    next_due_date = (SELECT next_due_date FROM maintenance_schedules WHERE id = $1),
    status = 'Active'
WHERE id = (SELECT equipment_id FROM maintenance_schedules WHERE id = $1);
💡 Every completed maintenance cycle auto-generates the next one.
